#include <stdio.h>

int main() {
    int num1, num2;
    int produto = 0;

    printf("--- Multiplicacao por Somas Sucessivas ---\n\n");

    // Leitura dos dois números inteiros
    printf("Digite o primeiro numero inteiro: ");
    scanf("%d", &num1);
    printf("Digite o segundo numero inteiro: ");
    scanf("%d", &num2);

    // Guarda o valor absoluto do segundo número para controlar o laço
    int vezes = num2;
    if (vezes < 0) {
        vezes = -vezes;
    }

    // Realiza as somas sucessivas
    for (int i = 0; i < vezes; i++) {
        produto += num1;
    }

    // Se o segundo número for negativo, inverte o sinal do resultado final
    if (num2 < 0) {
        produto = -produto;
    }

    // Exibe o resultado final
    printf("\nResultado de %d x %d = %d\n", num1, num2, produto);

    return 0;
}
