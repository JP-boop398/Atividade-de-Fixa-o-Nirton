#include <stdio.h>

int main() {
    int numero_secreto;
    int palpite;
    int acertou = 0; // Variável lógica (0 = falso, 1 = verdadeiro)

    printf("--- Jogo de Adivinhacao ---\n\n");

    // O Jogador 1 escolhe o número secreto com validação de intervalo
    do {
        printf("Jogador 1, escolha um numero entre 1 e 10: ");
        scanf("%d", &numero_secreto);
        
        if (numero_secreto < 1 || numero_secreto > 10) {
            printf("Numero invalido! Tente novamente.\n");
        }
    } while (numero_secreto < 1 || numero_secreto > 10);

    // Limpa a tela para o Jogador 2 não ver o número escolhido
    // (Funciona na maioria dos terminais)
    printf("\n\n\n\n\n\n\n\n\n\n\n\n\n\n\n\n\n\n\n\n");
    printf("--- Vez do Jogador 2 ---\n");

    // O Jogador 2 possui 4 chances
    for (int tentativa = 1; tentativa <= 4; tentativa++) {
        printf("\nTentativa %d de 4 - Digite seu palpite: ", tentativa);
        scanf("%d", &palpite);

        if (palpite == numero_secreto) {
            printf("\nParabens! Voce adivinhou o numero na %dª tentativa!\n", tentativa);
            acertou = 1; // Marca que o jogador ganhou
            break;       // Encerra o laço imediatamente
        } else {
            printf("Palpite incorreto!");
        }
    }

    // Se as 4 chances acabarem e ele não tiver acertado
    if (acertou == 0) {
        printf("\n\nVoce perdeu! Suas chances acabaram.");
        printf("\nO numero secreto era: %d\n", numero_secreto);
    }

    return 0;
}
