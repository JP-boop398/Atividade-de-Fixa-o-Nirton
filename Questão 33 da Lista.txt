#include <stdio.h>

int main() {
    float raio, area;
    const float PI = 3.141592;

    printf("--- Calculo da Area do Circulo ---\n\n");

    printf("Digite o valor do raio do circulo: ");
    scanf("%f", &raio);

    area = PI * raio * raio;

    printf("\n--- RESULTADO ---\n");
    printf("A area do circulo e: %.4f\n", area);

    return 0;
}
