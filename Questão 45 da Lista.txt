#include <stdio.h>

int main() {
    float razao, primeiro_termo, ultimo_termo;
    int n;
    float soma;

    printf("--- Soma dos Elementos de uma PA ---\n\n");

    printf("Digite a razao da PA: ");
    scanf("%f", &razao);

    printf("Digite o primeiro termo (a1): ");
    scanf("%f", &primeiro_termo);

    printf("Digite o ultimo termo (an): ");
    scanf("%f", &ultimo_termo);

    if (razao == 0) {
        printf("\nErro: A razao nao pode ser zero.\n");
        return 1;
    }

    n = (int)((ultimo_termo - primeiro_termo) / razao) + 1;

    if (n > 0) {
        soma = ((primeiro_termo + ultimo_termo) * n) / 2.0;

        printf("\n--- RESULTADOS ---\n");
        printf("Quantidade de termos encontrados (n): %d\n", n);
        printf("A soma de todos os elementos e: %.2f\n", soma);
    } else {
        printf("\nErro: Os dados inseridos nao formam uma PA valida.\n");
    }

    return 0;
}
