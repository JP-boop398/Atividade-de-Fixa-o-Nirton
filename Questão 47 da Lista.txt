#include <stdio.h>

int main() {
    int produtos_vendidos;
    float salario_fixo, comissao_por_produto, comissao_total, salario_final;

    float soma_salarios_pagos = 0.0;
    float maior_salario_fixo = -1.0;

    printf("--- Empresa Sempre Venda - Folha de Pagamento ---\n\n");

    for (int i = 1; i <= 1200; i++) {
        printf("--- Funcionario %d de 1200 ---\n", i);
        
        printf("Digite o salario fixo (R$): ");
        scanf("%f", &salario_fixo);

        printf("Digite a quantidade de produtos vendidos: ");
        scanf("%d", &produtos_vendidos);

        if (produtos_vendidos <= 5) {
            comissao_por_produto = 1.50;
        } else if (produtos_vendidos <= 50) {
            comissao_por_produto = 2.00;
        } else {
            comissao_por_produto = 2.50;
        }

        comissao_total = produtos_vendidos * comissao_por_produto;
        salario_final = salario_fixo + comissao_total;

        printf("-> Comissão total: R$ %.2f\n", comissao_total);
        printf("-> SALARIO FINAL: R$ %.2f\n\n", salario_final);

        soma_salarios_pagos += salario_final;

        if (salario_fixo > maior_salario_fixo) {
            maior_salario_fixo = salario_fixo;
        }
    }

    float media_salarios = soma_salarios_pagos / 1200.0;

    printf("=========================================\n");
    printf("--- RELATORIO GERAL DA EMPRESA ---\n");
    printf("-----------------------------------------\n");
    printf("b) Media dos salarios pagos: R$ %.2f\n", media_salarios);
    printf("c) Maior salario fixo encontrado: R$ %.2f\n", maior_salario_fixo);
    printf("=========================================\n");

    return 0;
}
