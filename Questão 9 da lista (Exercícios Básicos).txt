
#include <stdio.h>
#include <math.h>

int main()
{
    int num1, num2, resto;
    
    printf("Digite um número: ");
    scanf("%d", &num1);
    printf("Digite outro número: ");
    scanf("%d", &num2);
    
    resto = num1 % num2;
    
    printf("O resto da divisão do número %d com o número %d é: %d", num1, num2, resto);
    return 0;
}
