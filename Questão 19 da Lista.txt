#include <stdio.h>

int main() {
    float numero;
    int positivos = 0;

    printf("--- Contador de Numeros Positivos ---\n\n");

    // Loop para ler 5 números
    for (int i = 1; i <= 5; i++) {
        printf("Digite o %dº numero: ", i);
        scanf("%f", &numero);

        // Verifica se o número é maior que zero (positivo)
        if (numero > 0) {
            positivos++; // Incrementa o contador
        }
    }

    // Exibe o total de positivos encontrados
    printf("\nQuantidade de numeros positivos digitados: %d\n", positivos);

    return 0;
}
