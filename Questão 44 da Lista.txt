#include <stdio.h>

int main() {
    float nota1, nota2, nota3, media_final;

    printf("--- Calculo de Media Ponderada (Pesos 2, 3 e 5) ---\n\n");

    printf("Digite a primeira nota: ");
    scanf("%f", &nota1);

    printf("Digite a segunda nota: ");
    scanf("%f", &nota2);

    printf("Digite a terceira nota: ");
    scanf("%f", &nota3);

    media_final = ((nota1 * 2.0) + (nota2 * 3.0) + (nota3 * 5.0)) / 10.0;

    printf("\n--- RESULTADO FINAL ---\n");
    printf("Nota 1 (peso 2): %.1f\n", nota1);
    printf("Nota 2 (peso 3): %.1f\n", nota2);
    printf("Nota 3 (peso 5): %.1f\n", nota3);
    printf("------------------------\n");
    printf("MEDIA FINAL: %.2f\n", media_final);

    return 0;
}
