#include <stdio.h>

int main() {
    int numero;

    printf("--- Gerador de Tabuada (0 a 9) ---\n\n");


    printf("Digite um numero natural entre 0 e 9: ");
    scanf("%d", &numero);


    if (numero >= 0 && numero <= 9) {
        printf("\nTabuada do %d:\n", numero);
        

        for (int i = 1; i <= 10; i++) {
            printf("%d x %d = %d\n", numero, i, numero * i);
        }
    } else {
        printf("\nErro: O numero digitado nao esta entre 0 e 9.\n");
    }

    return 0;
}
