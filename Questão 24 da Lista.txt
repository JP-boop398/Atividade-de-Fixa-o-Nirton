#include <stdio.h>

int main() {
    float S = 0.0;
    int numerador = 50;
    int denominador = 1;

    // O laço roda 50 vezes, pois são 50 termos na soma
    for (int i = 1; i <= 50; i++) {
        // É preciso converter um dos números para float para a divisão não ser inteira
        S += (float)numerador / denominador;

        // Atualiza os valores para o próximo termo
        numerador--;
        denominador++;
    }

    // Exibe o resultado final do somatório
    printf("O valor do somatorio S e: %.4f\n", S);

    return 0;
}
