#include <stdio.h>

int main() {
    int A, B, C;
    float R, S, D;

    printf("--- Calculo da Expressao D ---\n\n");

    printf("Digite o valor de A (inteiro positivo): ");
    scanf("%d", &A);

    printf("Digite o valor de B (inteiro positivo): ");
    scanf("%d", &B);

    printf("Digite o valor de C (inteiro positivo): ");
    scanf("%d", &C);

    if (A >= 0 && B >= 0 && C >= 0) {
        // Cálculo de R = (A + B)²
        R = (A + B) * (A + B);

        S = (B + C) * (B + C);

        D = (R + S) / 2.0;

        printf("\n--- RESULTADOS ---\n");
        printf("Valor de R: %.1f\n", R);
        printf("Valor de S: %.1f\n", S);
        printf("Valor final de D: %.2f\n", D);
    } else {
        printf("\nErro: Por favor, digite apenas numeros inteiros positivos.\n");
    }

    return 0;
}
