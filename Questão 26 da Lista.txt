#include <stdio.h>

// Definição da função que imprime o padrão solicitado
void imprimirPadrao(int n) {
    // Laço externo controla as linhas (de 1 até n)
    for (int i = 1; i <= n; i++) {
        // Laço interno controla quantas vezes o número da linha se repete
        for (int j = 1; j <= i; j++) {
            printf("%d ", i);
        }
        // Quebra de linha após imprimir todos os números da linha atual
        printf("\n");
    }
}

int main() {
    int n;

    printf("Digite o valor de n: ");
    scanf("%d", &n);

    printf("\n--- Padrao Gerado ---\n");
    // Chamada da função passando o valor digitado pelo usuário
    imprimirPadrao(n);

    return 0;
}
