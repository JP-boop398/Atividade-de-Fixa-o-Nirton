#include <stdio.h>

int main() {
    float numero;

    printf("--- Algoritmo ECO (5 numeros) ---\n\n");

    // Loop que roda exatamente 5 vezes
    for (int i = 1; i <= 5; i++) {
        printf("Digite o %dº numero: ", i);
        scanf("%f", &numero);

        // Imprime imediatamente o numero lido (Efeito ECO)
        printf("ECO: %.2f\n\n", numero);
    }

    return 0;
}
